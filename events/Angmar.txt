add_namespace = angmar


# Angmar Demands Independence
country_event = {
	id = angmar.1
	title = angmar.1.t
	desc = angmar.1.d
	picture = GFX_report_event_paper
	
	is_triggered_only = yes
	
	option = { #Accept
		name = angmar.1.a
		ai_chance = {
			base = 50
			modifier = {
				has_opinion = { target = ANG value > 0 }
				add = 20
			}
			modifier = {
				has_war = yes
				add = 100
			}
		}
		GON = { end_puppet = ANG }
	}
	option = { #Refuse
		name = angmar.1.b
		ai_chance = {
			base = 50
			modifier = {
				add = 200
				has_government = fascism
			}
			modifier = {
				add = 30
				strength_ratio = { tag = ANG ratio > 1.0 }
			}
			modifier = {
				add = 50
				has_government = democratic
			}
		}
		GON = { end_puppet = ANG }
		GON = { declare_war_on = { target = ANG type = annex_everything } }
	}
}

# Communists Execute Coup (Angmar only)
country_event = {
	id = angmar.2
	title = political.3.t
	desc = political.3.d
	picture = GFX_report_event_gathering_fight
	
	trigger = {
		NOT = { has_government = communism }
		communism > 0.4
		is_major = no
		tag = ANG
		has_global_flag = ang_terrorism
	}

	mean_time_to_happen = {
		days = 100
		modifier = {
			factor = 0.5
			communism > 0.6
		}
	}

	immediate = {
	}
	
	option = {
		name = political.3.a
		set_politics = {
			ruling_party = communism
			elections_allowed = no
		}
		add_timed_idea = { idea = reign_of_terror days = 730 }
	}
}

	trigger = {
		NOT = { has_government = democratic }
		democratic > 0.4
		is_puppet = no
		tag = ANG
	}
	
	mean_time_to_happen = {
		days = 120
	}
	
	option = {
		name = political.17.a
		if = {
			limit = { has_government = fascism }
			add_timed_idea = { idea = fascism_defeated days = 730 }
		}
		if = {
			limit = { has_government = communism }
			add_timed_idea = { idea = communism_defeated days = 730 }
		}
		set_politics = {
			ruling_party = democratic
			elections_allowed = no
		}
	}
	option = {
		name = political.17.b
	}
}
